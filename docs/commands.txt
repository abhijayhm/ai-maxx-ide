--- Python server (Django) ---

# Activate venv (from repo root)
env\Scripts\activate

# First run / after model changes
cd server
python manage.py migrate

# Dev — ASGI (REST + WebSockets). daphne is first in INSTALLED_APPS, so runserver
# serves config.asgi.application (HTTP + WS). Do NOT pass --noasgi.
cd server
python manage.py runserver 127.0.0.1:8000
# Optional: raise Daphne WS frame limit if you stream very large single frames (default 1 MiB):
# daphne --websocket-max-message-size=16777216 config.asgi:application -b 127.0.0.1 -p 8000
# expect: "Starting ASGI/Daphne version ... development server at http://127.0.0.1:8000/"

# Production / tunnel (alternative to runserver):
cd server
..\env\Scripts\uvicorn.exe config.asgi:application --host 127.0.0.1 --port 8000

# If WS returns 404 + "No supported WebSocket library detected", reinstall:
..\env\Scripts\pip.exe install -r requirements.txt
# (needs websockets + wsproto from uvicorn[standard]; restart uvicorn after)

# Or from repo root with venv active:
uvicorn config.asgi:application --host 127.0.0.1 --port 8000 --app-dir server

# Tests
cd server
..\env\Scripts\python.exe -m pytest tests/ -v

--- Remote desktop (WebRTC) ---

# Server must run on the Windows machine whose screen you want to share.
# Installs aiortc + mss + av + pynput (see server/requirements.txt):
cd server
..\env\Scripts\pip install -r requirements.txt

# .env flags (sample.env):
#   REMOTE_INPUT_ENABLED=true   — allow trackpad/keyboard injection (pynput)
#   REMOTE_WEBRTC_STUB=false    — set true only for tests / no capture hardware

# Remote tab flow (Flutter):
#   1) WS connect  wss://{SERVER_DOMAIN}/api/ws/remote/?api_key=…&device_hash=…
#   2) Client sends WebRTC offer; server answers with desktop ScreenTrack (mss)
#   3) flutter_webrtc renders video; trackpad/keyboard use input_batch frames

# Flutter rebuild after adding flutter_webrtc:
cd app
C:\flutter\bin\flutter pub get
C:\flutter\bin\flutter run -d DEVICE_ID

# Verify remote WS + signaling (pytest uses REMOTE_WEBRTC_STUB=true):
cd server
..\env\Scripts\python.exe -m pytest tests/test_remote_ws.py -v


--- Cloudflare tunnel (server URL only) ---

# 1) One-time / rerun setup (Admin — double-click or cmd)
scripts\windows\setup_cloudflare_tunnel.bat

# Reads repo .env:
#   SERVER_DOMAIN=aimaxx.organisationapp.online
#   TUNNEL_NAME=ai-maxx-ide
#   BIND_PORT=8000

# 2) Start tunnel (separate terminal, after server is on 127.0.0.1:8000)
cloudflared tunnel run ai-maxx-ide

# Or background helper (starts tunnel if not connected)
scripts\windows\start_services.bat

# 3) Full dev stack (3 terminals)
#   Terminal A: uvicorn config.asgi:application --host 127.0.0.1 --port 8000  (in server/)
#   Terminal B: cloudflared tunnel run ai-maxx-ide
#   Terminal C: flutter run -d DEVICE_ID  (in app/)
